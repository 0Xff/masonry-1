$('#wrapper').masonry({

    singleMode: false,
    // disables measuring the width of each floated element.
    // Set to true if you the floated elements have the same width.
    // default: false

    columnWidth: 240,
    // width in pixels of 1 column of your grid
    // if unset, it defaults to the outer width of the first floated element

    itemSelector: '.box:visible',
    // additional selector to specify which elements inside
    // the wrapping element will be rearranged by masonry

    resizeable: true,
    // binds a Masonry call to window resizing
    // default: true

    appendedContent: $('.new_content'),
    // additional container selector for appended content
    // useful for Infinite Scroll integration

    saveOptions: true,
    // default: true
    // by default, Masonry will use the options from previous Masonry
    // calls, so you only have to enter in options once

},  function() {}
    // optional callback
    // 'this' will refer to the applicable elements masonry just rearranged

);
